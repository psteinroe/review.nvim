function calculateTotal(items) {
  let total = 0;
  for (let i = 0; i < items.length; i++) {
    total = total + items[i].price;
  }
  return total;
}

function processUser(user) {
  if (user.name == null) {
    return "Unknown";
  }
  return user.name.toUpperCase();
}
